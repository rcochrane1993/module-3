let hash = require('string-hash')

class firstHash {
  constructor() {
    this.list = []
  }

  get(x) {
    return this.list[hash(x)]
  }

  set(x, y) {
    this.list[hash(x)] = y
  }
}

let phoneSearch = new firstHash()
phoneSearch.set('Rob Cochrane',6317410828)
phoneSearch.set('Tim Law',3075468989)
phoneSearch.set('Matt Smith',5162893564)

console.log(phoneSearch.get('Rob Cochrane'))

let hash = require('string-hash')

class newspaperHash {
  constructor() {
    this.list = []
  }

  get(name,date) {
    return this.list[hash(name+date)]
  }

  set(paper) {
    this.list[hash(paper.name+paper.date)] = paper
  }
}

let paper = {name:'Newsday',date:'10/17/20011'}
let search = new newspaperHash()
search.set(paper)
search.get(paper.name,paper.date)

console.log(search.get(paper.name,paper.date))
